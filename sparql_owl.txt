# Query 1: List all books
SELECT ?book ?title
WHERE {
  ?book rdf:type ex:Book .
  ?book ex:title ?title .
}

# Query 2: List all users and their reading levels
SELECT ?user ?level
WHERE {
  ?user rdf:type ex:User .
  ?user ex:hasReadingLevel ?level .
}

# Query 3: Find all Science Fiction books
SELECT ?book ?title
WHERE {
  ?book rdf:type ex:Book .
  ?book ex:title ?title .
  ?book ex:hasTheme ex:ScienceFiction .
}

# Query 4: Find books suitable for Beginner readers
SELECT ?book ?title
WHERE {
  ?book rdf:type ex:Book .
  ?book ex:title ?title .
  ?book ex:suitableForLevel ex:Beginner .
}

# Query 5: Recommend books for Bob
SELECT ?book ?title
WHERE {
  ex:Bob ex:prefersTheme ?theme .
  ex:Bob ex:hasReadingLevel ?level .

  ?book rdf:type ex:Book .
  ?book ex:title ?title .
  ?book ex:hasTheme ?theme .
  ?book ex:suitableForLevel ?level .
